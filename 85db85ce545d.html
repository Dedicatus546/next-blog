function fn(n, arr) {
  const result = [];
  for (let i = 0; i < n; i++) {
    if (arr[i].length === 0 || arr[i].length === 1) {
      result.push(0);
      continue;
    }
    const strArr = arr[i].split("");
    const stack = [];
    stack[0] = strArr[0];
    let j = 1;
    let count = 0;
    while (j < strArr.length) {
      const peek = stack[stack.length - 1];
      if (peek === strArr[j]) {
        // æ¶ˆå»
        count++;
        stack.length--;
      } else {
        // pushè¿›æ•°ç»„
        stack[stack.length] = strArr[j];
      }
      j++;
    }
    result.push(count);
  }
  return result;
}
