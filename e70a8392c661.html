function fn(n, m, k, arr) {
    if (n === 0) {
        return -1;
    }
    const dp = [];
    dp[0] = arr[0];
    for (let i = 1; i < n; i++) {
        dp[i] = dp[i - 1] + arr[i];
    }
    let right = 10000000;
    for (let i = 0; i < n; i++) {
        for (let j = i; j < n; j++) {
            let avg = (dp[j] - dp[i] + arr[i]) / (j - i + 1);
            if (j - i + 1 >= m && avg >= k && j < right) {
                right = j;
            }
        }
    }
    return right === 100000000 ? -1 : right;
